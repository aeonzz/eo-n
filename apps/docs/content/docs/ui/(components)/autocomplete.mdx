## Usage

Import all parts and piece them together.

```tsx
import {
  Autocomplete,
  AutocompleteContent,
  AutocompleteEmpty,
  AutocompleteInput,
  AutocompleteItem,
  AutocompleteList,
} from "@components/ui/autocomplete";
```

```tsx
interface Fruit {
  id: string;
  value: string;
}

const fruits = [
  { id: "a", value: "Apple" },
  { id: "b", value: "Banana" },
  { id: "c", value: "Cherry" },
  { id: "d", value: "Durian" },
];

<Autocomplete items={fruits}>
  <AutocompleteInput />
  <AutocompleteContent>
    <AutocompleteEmpty>No fruits found.</AutocompleteEmpty>
    <AutocompleteList>
      {(fruit: Fruit) => (
        <AutocompleteItem key={fruit.id} value={fruit}>
          {fruit.value}
        </AutocompleteItem>
      )}
    </AutocompleteList>
  </AutocompleteContent>
</Autocomplete>;
```
