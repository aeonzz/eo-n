## Usage

Import all parts and piece them together.

```tsx
import {
  Toolbar,
  ToolbarButton,
  ToolbarGroup,
  ToolbarSeparator,
} from "@/components/ui/toolbar";
```

```tsx
<Toolbar>
  <ToolbarButton />
  <ToolbarSeparator />
  <ToolbarGroup>
    <ToolbarButton />
  </ToolbarGroup>
</Toolbar>
```
