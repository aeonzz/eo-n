## Usage

Import all parts and piece them together.

```tsx
import {
  Popover,
  PopoverContent,
  PopoverDescription,
  PopoverTitle,
  PopoverTrigger,
} from "@/components/ui/dialog";
```

```tsx
<Popover>
  <PopoverTrigger>Open</PopoverTrigger>
  <PopoverContent>
    <PopoverTitle>Popover Title</PopoverTitle>
    <PopoverDescription>
      This is some content inside the popover.
    </PopoverDescription>
  </PopoverContent>
</Popover>
```
