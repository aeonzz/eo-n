## Usage

Import all parts and piece them together.

```tsx
import {
  EmojiPicker,
  EmojiPickerContent,
  EmojiPickerSearch,
} from "@/components/ui/emoji-picker";
```

```tsx
<EmojiPicker
  onEmojiSelect={({ emoji }) => {
    console.log(emoji);
  }}
>
  <EmojiPickerSearch />
  <EmojiPickerContent />
</EmojiPicker>
```
